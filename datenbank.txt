drop table schueler;
drop table lehrer;
drop table klassen;
drop table faecher;
drop table tests;
drop table ergebnisse;

create table schueler(
	sid int not null,
	kid int,
	primary key(sid,kid),
	firstname varchar(20),
	lastname varchar(20),
	password varchar(256)
);

create table lehrer(
	lid int not null auto_increment,
	primary key(lid),
	firstname varchar(20),
	lastname varchar(20),
	password varchar(256)
);

create table klassen(
	kid int not null auto_increment,
	primary key(kid),
	klasse varchar(10)
);

create table faecher(
	fid int not null auto_increment,
        primary key(fid),
	fach varchar(20)
);

create table tests(
	tid int not null auto_increment,
	kid int,
	lid int,
	fid int,
	primary key(tid,kid,lid,fid),
	bezeichnung varchar(20),
	datum date	
);

create table ergebnisse(
	sid int,
	tid int,
	primary key(sid,tid),
	note int,
	kommentar varchar(20)
);
